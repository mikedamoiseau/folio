```rust
/// Update metadata fields on multiple books in a single transaction.
/// Only fields that are Some(...) are updated. Empty strings are stored as NULL.
pub fn bulk_update_metadata(
    conn: &Connection,
    ids: &[&str],
    author: Option<&str>,
    series: Option<&str>,
    publish_year: Option<u16>,
    language: Option<&str>,
    publisher: Option<&str>,
) -> Result<u32> {
    if ids.is_empty() {
        return Ok(0);
    }

    let mut set_clauses: Vec<String> = Vec::new();
    let mut param_values: Vec<Box<dyn rusqlite::types::ToSql>> = Vec::new();

    if let Some(v) = author {
        set_clauses.push("author = ?".to_string());
        param_values.push(Box::new(v.to_string()));
    }
    if let Some(v) = series {
        set_clauses.push("series = ?".to_string());
        let val: Option<String> = if v.is_empty() { None } else { Some(v.to_string()) };
        param_values.push(Box::new(val));
    }
    if let Some(v) = publish_year {
        set_clauses.push("publish_year = ?".to_string());
        let val: Option<u16> = if v == 0 { None } else { Some(v) };
        param_values.push(Box::new(val));
    }
    if let Some(v) = language {
        set_clauses.push("language = ?".to_string());
        let val: Option<String> = if v.is_empty() { None } else { Some(v.to_string()) };
        param_values.push(Box::new(val));
    }
    if let Some(v) = publisher {
        set_clauses.push("publisher = ?".to_string());
        let val: Option<String> = if v.is_empty() { None } else { Some(v.to_string()) };
        param_values.push(Box::new(val));
    }

    if set_clauses.is_empty() {
        return Ok(0);
    }

    // Add updated_at
    let now = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs() as i64;
    set_clauses.push("updated_at = ?".to_string());
    param_values.push(Box::new(now));

    let sql = format!("UPDATE books SET {} WHERE id = ?", set_clauses.join(", "));
    let tx = conn.unchecked_transaction()?;
    let mut count: u32 = 0;
    for id in ids {
        param_values.push(Box::new(id.to_string()));
        let params: Vec<&dyn rusqlite::types::ToSql> = param_values.iter().map(|p| p.as_ref()).collect();
        count += tx.execute(&sql, params.as_slice())? as u32;
        param_values.pop(); // remove the id for next iteration
    }
    tx.commit()?;
    Ok(count)
}
```

```rust
#[derive(serde::Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct BulkEditFields {
    pub author: Option<String>,
    pub series: Option<String>,
    pub publish_year: Option<u16>,
    pub language: Option<String>,
    pub publisher: Option<String>,
}

#[tauri::command]
pub async fn bulk_update_metadata(
    book_ids: Vec<String>,
    fields: BulkEditFields,
    state: State<'_, AppState>,
) -> Result<u32, String> {
    let conn = state.active_db()?.get().map_err(|e| e.to_string())?;
    let ids_ref: Vec<&str> = book_ids.iter().map(|s| s.as_str()).collect();

    let count = db::bulk_update_metadata(
        &conn,
        &ids_ref,
        fields.author.as_deref(),
        fields.series.as_deref(),
        fields.publish_year,
        fields.language.as_deref(),
        fields.publisher.as_deref(),
    )
    .map_err(|e| e.to_string())?;

    // Log activity
    log_activity(
        &conn,
        "bulk_edit",
        "book",
        None,
        None,
        Some(&format!("{} books updated", count)),
    );

    Ok(count)
}
```

- [ ] **Step 3: Implement BulkEditDialog**

Create `src/components/BulkEditDialog.tsx`:

```tsx
import { useState, useEffect } from "react";
import { invoke } from "@tauri-apps/api/core";
import { useTranslation } from "react-i18next";
import type { BookGridItem } from "../types";

interface BulkEditDialogProps {
  bookIds: string[];
  books: BookGridItem[];
  onClose: () => void;
  onSave: () => void;
}

interface FieldState {
  value: string;
  mixed: boolean;
  dirty: boolean;
}

function computeField(books: BookGridItem[], getter: (b: BookGridItem) => string | null | undefined): FieldState {
  const values = books.map(getter);
  const first = values[0] ?? "";
  const allSame = values.every(v => (v ?? "") === first);
  return { value: allSame ? first : "", mixed: !allSame, dirty: false };
}

export default function BulkEditDialog({ bookIds, books, onClose, onSave }: BulkEditDialogProps) {
  const { t } = useTranslation();
  const selected = books.filter(b => bookIds.includes(b.id));

  const [author, setAuthor] = useState<FieldState>(() => computeField(selected, b => b.author));
  const [series, setSeries] = useState<FieldState>(() => computeField(selected, b => b.series));
  const [year, setYear] = useState<FieldState>(() => computeField(selected, b => b.publish_year?.toString()));
  const [language, setLanguage] = useState<FieldState>(() => computeField(selected, b => b.language));
  const [publisher, setPublisher] = useState<FieldState>(() => computeField(selected, b => b.publisher));
  const [saving, setSaving] = useState(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const handleKey = (e: KeyboardEvent) => {
      if (e.key === "Escape") onClose();
    };
    document.addEventListener("keydown", handleKey);
    return () => document.removeEventListener("keydown", handleKey);
  }, [onClose]);

  const handleSave = async () => {
    const fields: Record<string, unknown> = {};
    if (author.dirty) fields.author = author.value;
    if (series.dirty) fields.series = series.value;
    if (year.dirty) {
      const parsed = parseInt(year.value);
      fields.publishYear = year.value === "" ? 0 : (isNaN(parsed) ? undefined : parsed);
    }
    if (language.dirty) fields.language = language.value;
    if (publisher.dirty) fields.publisher = publisher.value;

    if (Object.keys(fields).length === 0) {
      onClose();
      return;
    }

    setSaving(true);
    setError(null);
    try {
      await invoke("bulk_update_metadata", { bookIds, fields });
      onSave();
    } catch (e) {
      setError(String(e));
    } finally {
      setSaving(false);
    }
  };

  const fieldRow = (
    label: string,
    state: FieldState,
    setState: (s: FieldState) => void,
    type: "text" | "number" = "text",
  ) => (
    <div className="flex flex-col gap-1">
      <label className="text-xs text-ink-muted font-medium">{label}</label>
      <input
        type={type}
        value={state.dirty ? state.value : (state.mixed ? "" : state.value)}
        placeholder={state.mixed && !state.dirty ? t("bulkEdit.multipleValues") : undefined}
        onChange={(e) => setState({ value: e.target.value, mixed: state.mixed, dirty: true })}
        className={`h-9 px-3 bg-warm-subtle rounded-lg text-sm text-ink border border-transparent focus:border-accent focus:outline-none ${state.mixed && !state.dirty ? "italic text-ink-muted" : ""}`}
      />
    </div>
  );

  return (
    <>
      <div className="fixed inset-0 bg-ink/40 backdrop-blur-sm z-[80]" onClick={onClose} />
      <div className="fixed inset-0 z-[90] flex items-center justify-center p-4">
        <div
          role="dialog"
          aria-modal="true"
          className="bg-surface rounded-2xl shadow-2xl w-full max-w-md border border-warm-border p-6 space-y-4"
          onClick={(e) => e.stopPropagation()}
        >
          <h3 className="font-serif text-base font-semibold text-ink">
            {t("bulkEdit.title", { count: bookIds.length })}
          </h3>

          <div className="space-y-3">
            {fieldRow(t("bulkEdit.author"), author, setAuthor)}
            {fieldRow(t("bulkEdit.series"), series, setSeries)}
            {fieldRow(t("bulkEdit.year"), year, setYear, "number")}
            {fieldRow(t("bulkEdit.language"), language, setLanguage)}
            {fieldRow(t("bulkEdit.publisher"), publisher, setPublisher)}
          </div>

          {error && <p className="text-sm text-red-500">{error}</p>}

          <div className="flex gap-3 justify-end pt-1">
            <button
              type="button"
              onClick={onClose}
              className="px-4 py-2 text-sm text-ink-muted hover:text-ink transition-colors rounded-xl"
            >
              {t("common.cancel")}
            </button>
            <button
              type="button"
              onClick={handleSave}
              disabled={saving}
              className="px-4 py-2 text-sm bg-accent text-white rounded-xl hover:bg-accent-hover transition-colors font-medium disabled:opacity-50"
            >
              {saving ? t("common.saving") : t("common.save")}
            </button>
          </div>
        </div>
      </div>
    </>
  );
}
```
